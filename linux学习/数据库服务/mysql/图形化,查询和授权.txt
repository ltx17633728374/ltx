mysql 聚集函数
	avg(字段名)   #统计字段平均值
	sum(字段名)   #统计字段之和
	min(字段名)   #统计字段最小值
	max(字段名)    #统计字段最大值
	count(字段名)   #统计字段值个数

查询结果排序:
	order by 字段名  asc/desc ;     #升序排序/降序排序..默认是升序

查询结果分组:
	group by 字段名;    #不出现相同的组

限制查询结果显示行数:
	limit N;    #显示n行
	limit N,M;   #显示以第n-1行开始,显示m行
	where 条件 limit 3;   #显示前三行
	where 条件 limit 3,3;   #显示第四行到第七行

查询结果的过滤:
	having 条件表达式;
	where 条件 having 条件表达式;
	group by 字段名 having 条件表达式;
	
	select name from user where shell="/bin/bash" having name="mysql";
		查询登录器为shell的用户且过滤出名字为mysql的用户


复制表(做备份使用)
	create table db4.user1 select * from db3.user ;
	 	#把db3的user表所有内容复制到db4的user1
	create table db4.user2 select 	* from db3.user where 1=2;
		#复制表时添加不成立的条件,让复制时可以不复制内容,只复制表结构
 
多表查询:
	select 字段名列表 from 表a,表b;
	  #查出来的东西是表a乘以表b(笛卡尔集)
	select 字段名列表 from 表a,表b where 条件;
	select t2.name,t3.name from t2,t3 where t2.uid = t3.uid;
	#查询t2,t3表中uid相等的人的名字

where子查询:
	select name,uid from user2 where uid > (select avg(uid) from user2);
		  #查询user2表里面uid大于平均值的人的名字          
左连接查询
	select 字段名列表 from 表b left join 表b on 条件表达式;
	select * from t4 left join t5 on t4.uid=t5.uid
		#显示t4,t5表中符号条件的显示出来(显示以表4为主的)
右连接查询
	select 字段名列表 from 表a right join 表b on 条件表达式;

强制修改数据库密码:
	vim /etc/my.cnf
	skip-grant-tables
	#validate_password_policy=0
	#validate_password_length=1
	改完重启mysql
	直接无密码进入mysql里的mysql库
	select user,host,authentication_string from user
		#查找root的密码
	 update mysql.user
       set authentication_string=password("2222")
       where
       user="root" and host="localhost";
	 flush privileges;    #刷新策略
	#把root密码改为2222
	再次修改vim /etc/my.cnf
	#skip-grant-tables
	validate_password_policy=0
	validate_password_length=1
	
数据库授权:
	库名.表名(可以写具体的,写*.*时表示所有库所有表)
	客户端地址(可以写域名,ip,网段和表示所有主机的%)
	grant all on *.* to dba1@"%" identified by "2222" with grant option;
		#让所有客户端可以有所有权限用dba1用户登陆,with grant opition的意思是对其他用户授权
	set password=password("1234");
	#授权的用户改自己的密码
	set password for u1@"%"=password("1234")
	#管理员帮授权用户改他的密码 
权限撤销:
	use mysql;
	查表:
		user表,存储授权用户的访问权限
		db表,存储授权用户对数据库的访问权限
		tables_priv表,存储授权用户对表的访问权限
		columns_priv表,存储授权用户对字段的访问权限
	select user,host from mysql.user;  #查看所有授权用户	
	show grants for dba1@"%";
		#查看已经授权的用户的权限
	revoke grant option on *.* from dba1@"%";
		#删除dba1的授权权限
	revoke all on *.* from dba1@"%";
		#删除dba1的所有权限
	drop user root@"%";
		#删除用户
